#include<stdio.h>
#include<string.h> 
#include<stdlib.h> 
#define max 999
#define constyear 2024
struct in               //收入
{
	int year;  //年
	int month;  //月
	int day;    //日
    int money;     //金额   
 	char type[max];   //类别
 	char others[max];     //备注
}in[max];

struct out           //支出
{
	long year;  //年
	int month;  //月
	int day;    //日
    int money;     //金额   
 	char type[max];   //类别
 	char others[max];     //备注
}out[max];





int income(int *p1)  //---------------1记录输入账单
{
	long year;
	int month,day;
	printf("\n=================================\n");
	printf("欢迎使用个人账单管理系统\n");
	printf("=================================\n");
	printf("请输入收入信息:\n");
	
	printf("日期(YYYY-MM-DD):");
	fflush(stdin);
	while(1)
	{
		scanf("%4d-%2d-%2d",&year,&month,&day);
		if(year>0 && year<=constyear)  //年满足要求
		{
			if(month>0 && month<=12)  //月满足要求
			{
				if(day>0 && day<=31)//日满足要求
				{
					in[*p1].year=year;
					in[*p1].month=month;
					in[*p1].day=day;
					break;
				}
			}
		}
		printf("格式不正确!请重新输入！！！\n");
	}
	
	do
	{
		printf("金额(请输入正数):");
		scanf("%d",&in[*p1].money);
	}while(in[*p1].money<0);

	
	printf("类别(如工资、奖金等):");
	fflush(stdin);
	gets(in[*p1].type);
		
	printf("备注:");
	fflush(stdin);
	gets(in[*p1].others);
	
	(*p1)++;	
	
	printf("收入已成功记录！\n");
	
	printf("按任意键返回主菜单...\n");
	getchar();
	
	return 0;	
}

int outputIn(int temp[max],int *p1)  //输出收入

		a=temp[i];

//按照日期输出账单
void dataCheck(int *p1,int *p2)   
{
	int year,month,day;
	int count=0,count2=0,number[max],number2[max];     //记录多少个相同的日期
	
	
	printf("请输入日期(YYYY-MM-DD)：\n");
	while(1)
	{
		fflush(stdin);
		scanf("%4d-%2d-%2d",&year,&month,&day);
		if(year>0 && year<=constyear)  //年满足要求
		{
			if(month>0 && month<=12)  //月满足要求
			{
				if(day>0 && day<=31)//日满足要求
				{
					break;
				}
			}
		}
		printf("格式不正确! 请重新输入！！！\n");
	}
	
	
	//找收入
	for(int i=0;i<*p1;i++)   
	{
		
		if(in[i].year==year && in[i].month ==month)
		{
			if(in[i].day==day)
			{
				number[count]=i;
				count++;
			}	
		}	
    }
    outputIn(number,p1);
    
    //找支出
	for(int i=0;i<*p1;i++)   
	{
		if(out[i].year==year && out[i].month==month && out[i].day==day)
		{
			number2[count2]=i;
			count2++;
		}	
    }
    outputOut(number2,p2);
}

//按照类别输出账单

//----------4账单查询：用户可以按照指定日期、日期范围或类别查询账单，收入和支出分开显示
void checkBills(int *p1,int *p2)  
{
	int day,choice,temp1;

	do
	{
		printf("\n请输入要查询的账单的方式：\n");
		printf("1.指定日期\n");
		printf("2.日期范围\n");
		printf("3.类别\n");
		printf("0.返回主菜单（结束）\n");
		scanf("%d",&choice);
		switch(choice)
		{
			case 1:dataCheck(p1,p2);continue;
			case 2:dataArrangeCheck(p1,p2);continue;
			case 3:typeCheck(p1,p2);continue;
			case 0:break;
			default:printf("输入错误，请重新输入\n");continue;
		}
	}while(choice!=0);	
}



//------------3账单展示：系统能够展示所有录入的账单记录，收入和支出分开显示
void checkAllBills(int *p1,int *p2)  
{
	int temp[max],temp2[max];
	for(int i=0;i<*p1;i++)
	{
		temp[i]=i;
	}
	outputIn(temp,p1);
	
	for(int j=0;j<*p1;j++)
	{
		temp2[j]=j;
	}
	outputOut(temp2,p2);
}
